## ROUTE-BASED THEME SWITCHING

### Automatic Theme Detection

```typescript
// client/src/contexts/theme-context.tsx

export type VisualTheme = 'bold-minimaximalist' | 'mt-ocean';
export type DarkMode = 'light' | 'dark';

interface ThemeContextType {
  visualTheme: VisualTheme;
  darkMode: DarkMode;
  toggleDarkMode: () => void;
}

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const location = useLocation();
  const [darkMode, setDarkMode] = useState<DarkMode>('dark'); // Default dark
  
  // AUTO-DETECT VISUAL THEME BY ROUTE
  const visualTheme: VisualTheme = useMemo(() => {
    const marketingRoutes = ['/', '/marketing-prototype', '/pricing', '/about'];
    const isMarketing = marketingRoutes.includes(location[0]);
    
    return isMarketing ? 'bold-minimaximalist' : 'mt-ocean';
  }, [location]);
  
  // Apply theme to document
  useEffect(() => {
    document.documentElement.classList.toggle('dark', darkMode === 'dark');
    document.documentElement.setAttribute('data-theme', visualTheme);
    
    // Apply CSS custom properties
    const tokens = resolveTokens(visualTheme);
    applyCSSVariables(tokens);
  }, [darkMode, visualTheme]);
  
  const toggleDarkMode = () => setDarkMode(d => d === 'light' ? 'dark' : 'light');
  
  return (
    <ThemeContext.Provider value={{ visualTheme, darkMode, toggleDarkMode }}>
      {children}
    </ThemeContext.Provider>
  );
}

// Usage in components
function MyComponent() {
  const { visualTheme, darkMode } = useTheme();
  
  // Components automatically use correct theme via CSS variables
  return <Button className="btn-primary">Click Me</Button>;
}
```

```typescript
// client/src/components/adaptive/AdaptiveButton.tsx

import { useTheme } from '@/contexts/theme-context';
import { BoldButton } from '@/components/themed/bold/BoldButton';
import { OceanButton } from '@/components/themed/ocean/OceanButton';

export function AdaptiveButton(props: ButtonProps) {
  const { visualTheme } = useTheme();
  
  // Automatically use correct theme variant
  return visualTheme === 'bold-minimaximalist' 
    ? <BoldButton {...props} />
    : <OceanButton {...props} />;
}

// Usage: Just import AdaptiveButton everywhere
import { AdaptiveButton } from '@/components/adaptive/AdaptiveButton';

<AdaptiveButton>Submit</AdaptiveButton>

// On /feed â†’ Ocean turquoise glassmorphic button
```

```markdown
# Button Component

## Usage

\`\`\`tsx
import { AdaptiveButton } from '@/components/adaptive/AdaptiveButton';

<AdaptiveButton variant="primary">Submit</AdaptiveButton>
\`\`\`

## Variants

- `primary` - Main CTA (burgundy/turquoise based on theme)
- `secondary` - Secondary actions (outlined)
- `ghost` - Tertiary actions (transparent)

## Theme Behavior

### Bold Minimaximalist
- Background: #b91c3b (burgundy)
- Font weight: 600
- Border radius: 6px
```

## RISK MITIGATION

### Potential Risks

1. **Performance degradation** from glassmorphic effects
   - Mitigation: Hardware acceleration, selective blur usage
   
2. **Theme switching causing flicker**
   - Mitigation: Preload theme assets, CSS containment
   
3. **Accessibility violations** in custom components
   - Mitigation: Continuous automated testing, Layer 54 review
   
4. **Inconsistent token usage** across teams
   - Mitigation: ESLint rules, pre-commit hooks, documentation
   
5. **Visual regression on 82 pages**
   - Mitigation: Automated Playwright visual tests, Agent #A2
