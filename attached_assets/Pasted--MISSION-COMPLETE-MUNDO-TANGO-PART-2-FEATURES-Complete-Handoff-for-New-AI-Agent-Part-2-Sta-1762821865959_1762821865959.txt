// 1. Schema (shared/schema.ts)
export const founderApprovals = pgTable("founder_approvals", {
  id: serial("id").primaryKey(),
  contentType: varchar("content_type", { length: 50 }), // 'post', 'event', 'ai_response'
  contentId: integer("content_id").notNull(),
  aiGeneratedBy: varchar("ai_generated_by", { length: 100 }), // which AI agent
  content: jsonb("content"),
  status: varchar("status", { length: 20 }).default('pending'), // 'pending', 'approved', 'rejected'
  reviewedBy: integer("reviewed_by").references(() => users.id),
  reviewedAt: timestamp("reviewed_at"),
  notes: text("notes"),
  createdAt: timestamp("created_at").defaultNow(),
});

// 2. API Routes (server/routes/founderApprovalWorkflow.ts)
router.post('/api/admin/founder-approvals', async (req, res) => {
  const { contentType, contentId, content } = req.body;
  const approval = await db.insert(founderApprovals).values({...}).returning();
  // Send email notification to founder
  await sendEmail({
    to: founderEmail,
    subject: `New ${contentType} awaiting approval`,
    body: renderApprovalEmail(approval)
  });
  res.json(approval);
});

router.patch('/api/admin/founder-approvals/:id', async (req, res) => {
  const { status, notes } = req.body;
  const updated = await db.update(founderApprovals)
    .set({ status, notes, reviewedBy: req.user.id, reviewedAt: new Date() })
    .where(eq(founderApprovals.id, req.params.id))
    .returning();
  
  // If approved, publish content
  if (status === 'approved') {
    await publishContent(updated);
  }
  
  res.json(updated);
});

// 3. Frontend UI (client/src/pages/admin/FounderApprovals.tsx)
export default function FounderApprovalsPage() {
  const { data: approvals } = useQuery({
    queryKey: ['/api/admin/founder-approvals'],
  });
  
  return (
    <div className="p-6">
      <h1 className="text-3xl font-bold mb-6">Founder Approval Queue</h1>
      
      {approvals?.map(approval => (
        <ApprovalCard
          key={approval.id}
          approval={approval}
          onApprove={() => handleApprove(approval.id)}
          onReject={() => handleReject(approval.id)}
        />
      ))}
    </div>
  );
}
