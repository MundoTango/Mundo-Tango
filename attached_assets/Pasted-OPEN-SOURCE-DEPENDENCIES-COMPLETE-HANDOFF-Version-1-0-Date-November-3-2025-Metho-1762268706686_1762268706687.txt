```typescript
interface OpenSourceTracker {
```

```sql
CREATE TABLE open_source_packages (

  package_name VARCHAR(255) NOT NULL,
  current_version VARCHAR(50) NOT NULL,
  latest_version VARCHAR(50),
  license VARCHAR(100),
  category VARCHAR(50),
  vulnerabilities JSONB,
```
